/*a标签鼠标移入*/
a:hover { 
    border-color:white; /*边框颜色*/
    color:black; /*文字颜色*/
    background-color:white; /*背景变白*/
}

/*移除a标签样式*/
a:link,a:visited{
      text-decoration:none;
}


.tm-intro-link
{
    border:2px solid; /*边框*/
    color:white; /*边框及文字颜色*/
    /*border-radius:50px;圆角*/
    display:inline-block;
    padding:15px 60px;
}

/*渐变*/
a{
      transition:all 0.3s ease
}
